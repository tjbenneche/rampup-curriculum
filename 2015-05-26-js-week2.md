## Switch Statements

	var _day = "Tuesday";
    switch (_day) {
    	case "Monday":
        	alert("Today is Monday");
            break;
        case "Tuesday":
        	alert("Today is Tuesday");
            break;
        case "Wednesday":
        	alert("Today is Wednesday");
            break;
        case "Thursday":
        	alert("Today is Thursday");
            break;
        case "Friday":
        	alert("Today is Friday");
            break;
        case "Saturday":
        	alert("Today is Saturday");
            break;
        case "Sunday":
        	alert("Today is Sunday");
            break;
        default:
        	alert("That is not a day of the week");
    }
